53MEM_ERROR UARTWrapper::SendData(uint8_t *data, uint16_t *size, uint32_t timeout)
54{
55 if(!data || !size)
56 return MemoryErrorHandling::MEM_INVALID_ARGUMENT;
57
58 HAL_StatusTypeDef ret = HAL_UART_Transmit(&m_UARTHandle->m_UARTHandle, data, *size, timeout);
59 if(ret == HAL_BUSY)
60 return MemoryErrorHandling::MEM_HAL_BUSY;
61 if (ret == HAL_ERROR)
62 return MemoryErrorHandling::MEM_IO_ERROR;
63 if (ret == HAL_TIMEOUT)
64 return MemoryErrorHandling::MEM_HAL_TIMEOUT;
65
66 return MemoryErrorHandling::MEM_NO_ERROR;

108MEM_ERROR UARTWrapper::InitializeUARTDeviceSpecific(UARTHandle *uartProperties)
109{
110 int elemCtr = 0;
111 bool interfaceFound = false;
112 for(const AvailableUARTProperties& availPorts: m_AvailableUARTPorts)
113 {
114 if(availPorts.GetName() == uartProperties->m_InterfaceName)
115 {
116 uartProperties->m_UARTHandle.Instance = m_AvailableUARTPorts[elemCtr].GetUARTHandle();
117 interfaceFound = true;
118 if(uartProperties->m_Baudrate < availPorts.GetMinBaudrate() || uartProperties->m_Baudrate > availPorts.GetMaxBaudrate())
119 return MemoryErrorHandling::MEM_UNSUPPORTED_BAUDRATE;
120 break;
121 }
122 elemCtr++;
123 }
124
125 if(!interfaceFound)
126 return MemoryErrorHandling::MEM_INTERFACE_NOT_SUPPORTED;
127
128
129 uartProperties->m_UARTHandle.Init.BaudRate = uartProperties->m_Baudrate;
130
131 if (uartProperties->m_WordLength == UARTWrapper_WORD_LENGTH_5 ||
132 uartProperties->m_WordLength == UARTWrapper_WORD_LENGTH_6 ||
133 uartProperties->m_WordLength == UARTWrapper_WORD_LENGTH_7)
134 return MemoryErrorHandling::MEM_INVALID_ARGUMENT;
135 else if (uartProperties->m_WordLength == UARTWrapper_WORD_LENGTH_8)
136 uartProperties->m_UARTHandle.Init.WordLength = UART_WORDLENGTH_8B;
137 else if (uartProperties->m_WordLength == UARTWrapper_WORD_LENGTH_9)
138 uartProperties->m_UARTHandle.Init.WordLength = UART_WORDLENGTH_9B;
139
140 if(uartProperties->m_StopBits == UARTWrapper_STOP_BITS_1)
141 uartProperties->m_UARTHandle.Init.StopBits = UART_STOPBITS_1;
142 else if(uartProperties->m_StopBits == UARTWrapper_STOP_BITS_2)
143 uartProperties->m_UARTHandle.Init.StopBits = UART_STOPBITS_2;
144
145 if(uartProperties->m_Parity == UARTWrapper_NO_PARITY)
146 uartProperties->m_UARTHandle.Init.Parity = UART_PARITY_NONE;
147 else if(uartProperties->m_Parity == UARTWrapper_PARITY_EVEN)
148 uartProperties->m_UARTHandle.Init.Parity = UART_PARITY_EVEN;
149 else if(uartProperties->m_Parity == UARTWrapper_PARITY_ODD)
150 uartProperties->m_UARTHandle.Init.Parity = UART_PARITY_ODD;
151
152 if (uartProperties->m_Mode == UARTWrapper_TRANSMIT)
153 uartProperties->m_UARTHandle.Init.Mode = UART_MODE_TX;
154 if (uartProperties->m_Mode == UARTWrapper_RECEIVE)
155 uartProperties->m_UARTHandle.Init.Mode = UART_MODE_RX;
156 if (uartProperties->m_Mode == UARTWrapper_TRANSMIT_RECEIVE)
157 uartProperties->m_UARTHandle.Init.Mode = UART_MODE_TX_RX;
158
159 uartProperties->m_UARTHandle.Init.HwFlowCtl = UART_HWCONTROL_NONE;
160 uartProperties->m_UARTHandle.Init.OverSampling = UART_OVERSAMPLING_16;
161 uartProperties->m_UARTHandle.gState = HAL_UART_STATE_RESET;
162 if (HAL_UART_Init(&uartProperties->m_UARTHandle) != HAL_OK)
163 return MemoryErrorHandling::MEM_HAL_INTERNAL_ERROR;
164
165 return MemoryErrorHandling::MEM_NO_ERROR;
