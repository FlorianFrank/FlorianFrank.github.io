167MEM_ERROR SPIWrapper::SendData(uint8_t *data, uint16_t *size, uint32_t timeoutInMs)
168{
169 if(!size || *size == 0)
170 return MemoryErrorHandling::MEM_INVALID_ARGUMENT;
171
172 bool retCS = ReadChipSelect();
173
174 TimeMeasurement tm{};
175 tm.ResetAndStartTimer();
176 while(!retCS)
177 {
178 if(tm.GetElapsedTimeInMS())
179 return MemoryErrorHandling::MEM_HAL_TIMEOUT;
180 retCS = ReadChipSelect();
181 }
182
183 bool writeEnable = false;
184 MEM_ERROR ret;
185 if(m_SPIHandle->m_Mode == SPI_MASTER)
186 SetChipSelect();
187 else
188 {
189 while(!writeEnable)
190 {
191 writeEnable = ReadChipSelect();
192 if(tm.GetElapsedTimeInMS() >= timeoutInMs)
193 return MemoryErrorHandling::MEM_HAL_TIMEOUT;
194 }
195 }
196#if STM32
197 ret = MemoryErrorHandling::HAL_StatusTypeDefToErr(
198 HAL_SPI_Transmit(&m_SPIHandle->m_SPIHandle, data, *size, timeoutInMs));
199#else
200 // m_SPIHandle.StoreBuffer(data, size); TODO
201#endif // STM32
202 if(m_SPIHandle->m_Mode == SPI_MASTER)
203 ResetChipSelect();
204 return ret;
205}

214MEM_ERROR SPIWrapper::ReceiveData(uint8_t *data, uint16_t *size, uint32_t timeoutInMs)
215{
216 if(!size || *size == 0)
217 return MemoryErrorHandling::MEM_INVALID_ARGUMENT;
218
219 bool retCS = ReadChipSelect();
220 TimeMeasurement tm{};
221 uint32_t timeStampStart = tm.ResetAndStartTimer();
222 while(!retCS && tm.GetElapsedTimeInMS() - timeStampStart < timeoutInMs)
223 {
224 retCS = ReadChipSelect();
225 }
226
227
228 MEM_ERROR ret;
229 if(m_SPIHandle->m_Mode == SPI_MASTER)
230 SetChipSelect();
231 else
232 {
233 bool readEnable = false;
234 TimeMeasurement tm;
235 uint32_t timeStampStart = tm.ResetAndStartTimer();
236 while(!readEnable)
237 {
238 readEnable = ReadChipSelect();
239 if(tm.GetElapsedTimeInMS() >= timeoutInMs)
240 return MemoryErrorHandling::MEM_HAL_TIMEOUT;
241 }
242 }
243
244#if STM32
245 ret = MemoryErrorHandling::HAL_StatusTypeDefToErr(HAL_SPI_Receive(&m_SPIHandle->m_SPIHandle, data, *size, timeoutInMs));
246#else
247 //m_SPIHandle.ReadBuffer(data, size); TODO
248 ret = MemoryErrorHandling::MEM_NO_ERROR;
249#endif // STM32
250 if(m_SPIHandle->m_Mode == SPI_MASTER)
251 ResetChipSelect();
252 return ret;
253}
