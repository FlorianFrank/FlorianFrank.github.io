137MemoryErrorHandling::MEM_ERROR STM32F429Wrapper::DeInitialize()
138{
139 while(!m_InitializedGPIOPins.empty())
140 {
141 GPIOPin pin = m_InitializedGPIOPins.back();
142 auto ret = DeInitializeGPIOPin(pin);
143 if(ret != MemoryErrorHandling::MEM_NO_ERROR)
144 return ret;
145 }
146
147 while(!m_InitializedGPIOBanks.empty())
148 {
149 GPIOBank bank = m_InitializedGPIOBanks.back();
150 auto ret = DeInitializeGPIOBank(bank);
151 if(ret != MemoryErrorHandling::MEM_NO_ERROR)
152 return ret;
153 }
154
155 MemoryErrorHandling::MEM_ERROR err = MemoryErrorHandling::HAL_StatusTypeDefToErr(HAL_DeInit());
156 if(err != MemoryErrorHandling::MEM_NO_ERROR)
157 return err;
158
159 // Disable SYS_CLK
160 __HAL_RCC_PWR_CLK_DISABLE();
161
162 return MemoryErrorHandling::MEM_NO_ERROR;
163}

216MemoryErrorHandling::MEM_ERROR STM32F429Wrapper::InitializeGPIOPin(GPIOPin gpioPin, GPIO_PinState gpioPinState, uint32_t mode, uint32_t pull,
217 uint32_t speed, uint32_t alternate)
218{
219 GPIOBank bank = gpioPin.GetGPIOBank();
220 if (!bank)
221 return MemoryErrorHandling::MEM_INVALID_ARGUMENT;
222
223 if(std::find(m_InitializedGPIOBanks.begin(), m_InitializedGPIOBanks.end(), gpioPin.GetGPIOBank()) == m_InitializedGPIOBanks.end())
224 {
225 MemoryErrorHandling::MEM_ERROR ret = InitializeGPIOBank(gpioPin.GetGPIOBank());
226 if(ret != MemoryErrorHandling::MEM_NO_ERROR)
227 return ret;
228 m_InitializedGPIOBanks.push_back(bank);
229 }
230
231 if(std::find(m_InitializedGPIOPins.begin(), m_InitializedGPIOPins.end(), gpioPin) != m_InitializedGPIOPins.end())
232 return MemoryErrorHandling::MEM_GPIO_PIN_ALREADY_IN_USE;
233
234 uint32_t pin = gpioPin.GetGPIOPin();
235 GPIO_InitTypeDef gpioInitStruct = {0};
236 gpioInitStruct.Pin = pin;
237 gpioInitStruct.Mode = mode;
238 gpioInitStruct.Pull = pull;
239 gpioInitStruct.Speed = speed;
240 gpioInitStruct.Alternate = alternate;
241
242 HAL_GPIO_Init(bank, &gpioInitStruct);
243 HAL_GPIO_WritePin(bank, pin, gpioPinState);
244 m_InitializedGPIOPins.push_back(gpioPin);
245
246 return MemoryErrorHandling::MEM_NO_ERROR;
247}

255MemoryErrorHandling::MEM_ERROR STM32F429Wrapper::DeInitializeGPIOPin(const GPIOPin &gpioPin)
256{
257 HAL_GPIO_DeInit(gpioPin.GetGPIOBank(), gpioPin.GetGPIOPin());
258 if(std::find(m_InitializedGPIOPins.begin(), m_InitializedGPIOPins.end(), gpioPin) != m_InitializedGPIOPins.end())
259 m_InitializedGPIOPins.remove(gpioPin);
260 else
261 return MemoryErrorHandling::MEM_HAL_INTERNAL_ERROR;
262
263 m_InitializedGPIOPins.erase(std::remove(m_InitializedGPIOPins.begin(), m_InitializedGPIOPins.end(), gpioPin),
264 m_InitializedGPIOPins.end());
265
266 return MemoryErrorHandling::MEM_NO_ERROR;
