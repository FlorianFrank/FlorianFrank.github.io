197MEM_ERROR MemoryControllerSPI::SetWriteEnableLatch(bool checkRegister, uint32_t timeoutInMs)
198{
199 SPIWrapper::SetWriteProtect();
200 MEM_ERROR ret = SendSPICommand(ReRAM_WREN, nullptr, false);
201 SPIWrapper::ResetWriteProtect();
202 if(ret != MemoryErrorHandling::MEM_NO_ERROR)
203 return ret;

205 if(checkRegister)

246uint32_t MemoryControllerSPI::PollWriteInProgressRegister(uint32_t timeoutCycles)
247{
248 uint32_t endTS = 0;
249
250 //start timer
251 TimeMeasurement tm{};
252 uint32_t startTS = tm.ResetAndStartTimer();
253 do {
254 MemoryStatusRegister statusRegister;
255 MEM_ERROR err = ReadStatusRegister(statusRegister);
256 if(err != MemoryErrorHandling::MEM_NO_ERROR )
257 return err;
258
259 if (statusRegister.WriteInProgressBit == 0) {
260 return tm.StopTimer();
261 }
262 if(timeoutCycles != 0)
263 endTS =tm.GetTimer();
264 }while(startTS + timeoutCycles > endTS);
265
266 return endTS;
267}
